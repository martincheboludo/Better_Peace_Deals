
BPD_Handle_Vichy_France = {
		if = {
				limit = {
				
				original_tag = GER
				
				any_country_with_original_tag = {
					original_tag_to_check = FRA
					has_government = fascism
					OR = {
						is_subject_of = GER
						has_focus_tree = vichy_french_focus
					}
					NOT = { has_war_with = GER }
					exists = yes
				}
				
				country_exists = FRA
				FRA = { is_in_faction_with = FROM }
				FRA = { has_war_with = ROOT }
			
			}
				random_other_country = {
					limit = {
						original_tag = FRA
						has_government = fascism
						OR = {
							is_subject_of = GER
							has_focus_tree = vichy_french_focus
						}
					}
					annex_country = {
						target = FRA
						transfer_troops = yes
					}
					every_owned_state = {
						limit = { 
							OR = {
								is_claimed_by = GER
								is_core_of = GER
							}
						} 
						GER = { 
							transfer_state = PREV 
						} 
					}
				}
			}
}
